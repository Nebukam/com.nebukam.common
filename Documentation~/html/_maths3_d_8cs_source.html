141 //Finds the intersection points between a straight line and a spline. Solves a Cubic polynomial equation
142 //The output is in the form of a percentage along the length of the spline (range 0 to 1).
143 //The linePoints array should contain two points which form a straight line.
144 //The cPoints array should contain all the control points of the spline.
145 //Use case: create a gauge with a non-linear scale by defining an array with needle angles vs the number it should point at. The array creates a spline.
146 //Driving the needle with a float in range 0 to 1 gives an unpredictable result. Instead, use the GetLineSplineIntersections() function to find the angle the
147 //gauge needle should have for a given number it should point at. In this case, cPoints should contain x for angle and y for scale number.
148 //Make a horizontal line at the given scale number (y) you want to find the needle angle for. The returned float is a percentage location on the spline (range 0 to 1).
149 //Plug this value into the GetPointOnSpline() function to get the x coordinate which represents the needle angle.
150 //Source: https://medium.com/@csaba.apagyi/finding-catmull-rom-spline-and-line-intersection-part-2-mathematical-approach-dfb969019746
151 public static float[] GetLineSplineIntersections(Vector2[] linePoints, Vector2[] cPoints)
152 {
153
154 List<float> list = new List<float>();
155 float[] crossings;
156
157 int numSections = cPoints.Length - 3;
158
159 //The line spline intersection can only be calculated for one segment of a spline, meaning 4 control points,
160 //with a spline segment between the middle two control points. So check all spline segments.
161 for (int i = 0; i < numSections; i++)
162 {
163
164 //Get the 4 control points around the location to be sampled.
165 Vector2 p0 = cPoints[i];
166 Vector2 p1 = cPoints[i + 1];
167 Vector2 p2 = cPoints[i + 2];
168 Vector2 p3 = cPoints[i + 3];
169
170 //The Catmull-Rom spline can be written as:
171 // 0.5 * (2P1 + (-P0 + P2) * t + (2P0 - 5P1 + 4P2 - P3) * t^2 + (-P0 + 3P1 - 3P2 + P3) * t^3)
172 //Variables P0 to P3 are the control points.
173 //Notation: 2P1 means 2*controlPoint1
174 //Variable t is the position on the spline, converted from a range of 0 to 1.
175 //C# way of writing the function is below. Note that f means float (to force precision).
176 //Vector2 result = .5f * (2f * p1 + (-p0 + p2) * t + (2f * p0 - 5f * p1 + 4f * p2 - p3) * (t * t) + (-p0 + 3f * p1 - 3f * p2 + p3) * (t * t * t));
177
178 //The variable t is the only unknown, so the rest can be substituted:
179 //a = 0.5 * (-p0 + 3*p1 - 3*p2 + p3)
180 //b = 0.5 * (2*p0 - 5*p1 + 4*p2 - p3)
181 //c = 0.5 * (-p0 + p2)
182 //d = 0.5 * (2*p1)
183
184 //This gives rise to the following Cubic equation:
185 //a * t^3 + b * t^2 + c * t + d = 0
186
187 //The spline control points (p0-3) consist of two variables: the x and y coordinates. They are independent so we can handle them separately.
188 //Below, a1 is substitution a where the x coordinate of each point is used, like so: a1 = 0.5 * (-p0.x + 3*p1.x - 3*p2.x + p3.x)
189 //Below, a2 is substitution a where the y coordinate of each point is used, like so: a2 = 0.5 * (-p0.y + 3*p1.y - 3*p2.y + p3.y)
190 //The same logic applies for substitutions b, c, and d.
191
192 float a1 = 0.5f * (-p0.x + 3f * p1.x - 3f * p2.x + p3.x);
193 float a2 = 0.5f * (-p0.y + 3f * p1.y - 3f * p2.y + p3.y);
194 float b1 = 0.5f * (2f * p0.x - 5f * p1.x + 4f * p2.x - p3.x);
195 float b2 = 0.5f * (2f * p0.y - 5f * p1.y + 4f * p2.y - p3.y);
196 float c1 = 0.5f * (-p0.x + p2.x);
197 float c2 = 0.5f * (-p0.y + p2.y);
198 float d1 = 0.5f * (2f * p1.x);
199 float d2 = 0.5f * (2f * p1.y);
200
201 //We now have two Cubic functions. One for x and one for y.
202 //Note that a, b, c, and d are not vector variables itself but substituted functions.
203 //x = a1 * t^3 + b1 * t^2 + c1 * t + d1
204 //y = a2 * t^3 + b2 * t^2 + c2 * t + d2
205
206 //Line formula, standard form:
207 //Ax + By + C = 0
208 float A = linePoints[0].y - linePoints[1].y;
209 float B = linePoints[1].x - linePoints[0].x;
210 float C = (linePoints[0].x - linePoints[1].x) * linePoints[0].y + (linePoints[1].y - linePoints[0].y) * linePoints[0].x;
211
212 //Substituting the values of x and y from the separated Spline formula into the Line formula, we get:
213 //A * (a1 * t^3 + b1 * t^2 + c1 * t + d1) + B * (a2 * t^3 + b2 * t^2 + c2 * t + d2) + C = 0
214
215 //Rearranged version:
216 //(A * a1 + B * a2) * t^3 + (A * b1 + B * b2) * t^2 + (A * c1 + B * c2) * t + (A * d1 + B * d2 + C) = 0
217
218 //Substituting gives rise to a Cubic function:
219 //a * t^3 + b * t^2 + c * t + d = 0
220 float a = A * a1 + B * a2;
221 float b = A * b1 + B * b2;
222 float c = A * c1 + B * c2;
223 float d = A * d1 + B * d2 + C;

226 //This is again a Cubic equation, combined from the Line and the Spline equation. If you solve this you can get up to 3 line-spline cross points.
227 //How to solve a Cubic equation is described here:
228 //https://www.cs.rit.edu/~ark/pj/lib/edu/rit/numeric/Cubic.shtml
229 //https://www.codeproject.com/Articles/798474/To-Solve-a-Cubic-Equation

232 float cross1;
233 float cross2;
234 float cross3;
235 float crossCorrected;

238 // SolveCubic2(out crossAmount, out cross1, out cross2, out cross3, a, b, c, d);
239 SolveCubic(out crossAmount, out cross1, out cross2, out cross3, a, b, c, d);
240
241 //Get the highest and lowest value (in range 0 to 1) of the current section and calculate the difference.
242 float currentSectionLowest = (float)i / (float)numSections;
243 float currentSectionHighest = ((float)i + 1f) / (float)numSections;
244 float diff = currentSectionHighest - currentSectionLowest;
245
246 //Only use the result if it is within range 0 to 1.
247 //The range 0 to 1 is within the current segment. It has to be converted to the range of the entire spline,
248 //which still uses a range of 0 to 1.
249 if (cross1 >= 0 && cross1 <= 1)
250 {
251
252 //Map an intermediate range (0 to 1) to the lowest and highest section values.
253 crossCorrected = (cross1 * diff) + currentSectionLowest;
254
255 //Add the result to the list.
256 list.Add(crossCorrected);
257 }
258
259 if (cross2 >= 0 && cross2 <= 1)
260 {
261
262 //Map an intermediate range (0 to 1) to the lowest and highest section values.
263 crossCorrected = (cross2 * diff) + currentSectionLowest;
264
265 //Add the result to the list.
266 list.Add(crossCorrected);
267 }
268
269 if (cross3 >= 0 && cross3 <= 1)
270 {
271
272 //Map an intermediate range (0 to 1) to the lowest and highest section values.
273 crossCorrected = (cross3 * diff) + currentSectionLowest;
274
275 //Add the result to the list.
276 list.Add(crossCorrected);
277 }
278 }
279
280 //Convert the list to an array.
281 crossings = list.ToArray();
282
283 return crossings;

286 //Solve cubic equation according to Cardano.
287 //Source: https://www.cs.rit.edu/~ark/pj/lib/edu/rit/numeric/Cubic.shtml
288 private static void SolveCubic(out int nRoots, out float x1, out float x2, out float x3, float a, float b, float c, float d)
289 {
290
291 float TWO_PI = 2f * Mathf.PI;
292 float FOUR_PI = 4f * Mathf.PI;
293
294 // Normalize coefficients.
295 float denom = a;
296 a = b / denom;
297 b = c / denom;
298 c = d / denom;
299
300 // Commence solution.
301 float a_over_3 = a / 3f;
302 float Q = (3f * b - a * a) / 9f;
303 float Q_CUBE = Q * Q * Q;
304 float R = (9f * a * b - 27f * c - 2f * a * a * a) / 54f;
305 float R_SQR = R * R;
306 float D = Q_CUBE + R_SQR;
307
308 if (D < 0.0f)
309 {
310
311 // Three unequal real roots.
312 nRoots = 3;
313 float theta = Mathf.Acos(R / Mathf.Sqrt(-Q_CUBE));
314 float SQRT_Q = Mathf.Sqrt(-Q);
315 x1 = 2f * SQRT_Q * Mathf.Cos(theta / 3f) - a_over_3;
316 x2 = 2f * SQRT_Q * Mathf.Cos((theta + TWO_PI) / 3f) - a_over_3;
317 x3 = 2f * SQRT_Q * Mathf.Cos((theta + FOUR_PI) / 3f) - a_over_3;
318 }
319
320 else if (D > 0.0f)
321 {
322
323 // One real root.
324 nRoots = 1;
325 float SQRT_D = Mathf.Sqrt(D);
326 float S = CubeRoot(R + SQRT_D);
327 float T = CubeRoot(R - SQRT_D);
328 x1 = (S + T) - a_over_3;
329 x2 = float.NaN;
330 x3 = float.NaN;
331 }
332
333 else
334 {
335
336 // Three real roots, at least two equal.
337 nRoots = 3;
338 float CBRT_R = CubeRoot(R);
339 x1 = 2 * CBRT_R - a_over_3;
340 x2 = CBRT_R - a_over_3;
341 x3 = x2;
342 }
343 }
