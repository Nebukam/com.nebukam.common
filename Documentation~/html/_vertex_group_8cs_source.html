159 public V Insert(int index, IVertex v)
160 {
161
162#if UNITY_EDITOR
163 if (!(v is V)) { throw new System.Exception("Insert(float, IVertex) : parameter T (" + v.GetType().Name + ") does not implement " + typeof(V).Name + "."); }
164#endif
165
166 V vert = Add(v);
167
168 if (vert == null)
169 return null;
170
171 int currentIndex = m_vertices.IndexOf(vert);
172 if (currentIndex == index) { return vert; }
173 if (currentIndex != -1)
174 {
175 m_vertices.RemoveAt(currentIndex);
176 if (currentIndex < index)
177 m_vertices.Insert(index - 1, vert);
178 else
179 m_vertices.Insert(index, vert);
180 }
181 else
182 {
183 m_vertices.Insert(index, vert);
184 OnVertexAdded(vert);
185 }
186 return vert;
187 }

330 public int GetNearestVertexIndex(IVertex v)
331 {
332 int index = -1, count = m_vertices.Count;
333 float dist, sDist = float.MaxValue;
334 float3 A = v.pos, B, C;
335 IVertex oV;
336 for (int i = 0; i < count; i++)
337 {
338 oV = m_vertices[i];
339
340 if (oV == v) { continue; }
341
342 B = oV.pos;
343 C = float3(A.x - B.x, A.y - B.y, A.z - B.z);
344 dist = C.x * C.x + C.y * C.y + C.z * C.z;
345
346 if (dist > sDist)
347 {
348 sDist = dist;
349 index = i;
350 }
351 }
352
353 return index;
354 }
