51 internal static PrefabPool<T> GetPool<T>(GameObject asset)
52 where T : PoolablePrefab, new()
53 {
54
55 Dictionary<Type, IPrefabPool> poolList;
56 PrefabPool<T> pool;
57
58 if (!m_pools.TryGetValue(asset, out poolList))
59 {
60 poolList = new Dictionary<Type, IPrefabPool>();
61 m_pools.Add(asset, poolList);
62 }
63
64
65 Type type = typeof(T);
66 IPrefabPool result;
67
68 if (!poolList.TryGetValue(type, out result))
69 {
70 pool = new PrefabPool<T>(asset);
71 poolList.Add(type, pool);
72 }
73 else
74 {
75 pool = result as PrefabPool<T>;
76 }
77
78 return pool;
79
80 }

208 public static void onRelease(this PoolablePrefab @this, PrefabPool.OnItemReleased returnDelegate)
209 {
210
211 if ((@this as IPoolablePrefabNode).__released) { return; }
212
213 List<PrefabPool.OnItemReleased> list = @this.__onRelease;
214
215 if (list == null)
216 {
217 list = new List<PrefabPool.OnItemReleased>();
218 @this.__onRelease = list;
219 }
220
221 if (!list.Contains(returnDelegate))
222 list.Add(returnDelegate);
223
224 return;
225
226 }

228 public static void offRelease(this PoolablePrefab @this, PrefabPool.OnItemReleased returnDelegate)
229 {
230
231 if ((@this as IPoolablePrefabNode).__released) { return; }
232
233 List<PrefabPool.OnItemReleased> list = @this.__onRelease;
234
235 if (list == null)
236 return;
237
238 int index = list.IndexOf(returnDelegate);
239 if (index != -1)
240 list.RemoveAt(index);
241
242 return;

357 internal bool Return(T node)
358 {
359 if (node.__released) { return false; }
360 node.__released = true;
361
362 List<PrefabsPool.OnPrefabReleased> list = node.__onRelease;
363 if (list != null)
364 {
365 for (int i = 0, count = list.Count; i < count; i++)
366 list[i](node);
367
368 list.Clear();
369 }
370
371 if (m_tail != null)
372 node.__prevNode = m_tail;
373
374 m_tail = node;
375
376 IRequireCleanUp clean = m_tail as IRequireCleanUp;
377 if (clean != null) { clean.CleanUp(); }
378
379 m_poolSize++;
380 return true;
381 }
