352 public static bool IsBetween(float2 a, float2 b, float2 c)
353 {
354
355 float2 ab = float2(b.x - a.x, b.y - a.y);//Entire line segment
356 float2 ac = float2(c.x - a.x, c.y - a.y);//The intersection and the first point
357
358 float dot = ab.x * ac.x + ab.y * ac.y;
359
360 //If the vectors are pointing in the same direction = dot product is positive
361 if (dot <= 0f) { return false; }
362
363 float abm = ab.x * ab.x + ab.y * ab.y;
364 float acm = ac.x * ac.x + ac.y * ac.y;
365
366 //If the length of the vector between the intersection and the first point is smaller than the entire line
367 return (abm >= acm);
368 }

455 public static bool TryGetCircleIntersection(
456 float2 centerA,
457 float radiusA,
458 float2 centerB,
459 float radiusB,
460 out float2 pt1,
461 out float2 pt2)
462 {
463
464 float2 AB = float2(centerA.x - centerB.x, centerA.y - centerB.y);
465 float d = sqrt(AB.x * AB.x + AB.y * AB.y);
466
467 if (d <= (radiusA + radiusB) && d >= abs(radiusB - radiusA))
468 {
469
470 float ex = (centerB.x - centerA.x) / d, ey = (centerB.y - centerA.y) / d,
471 x = (radiusA * radiusA - radiusB * radiusB + d * d) / (2 * d),
472 y = sqrt(radiusA * radiusA - x * x),
473 xex = centerA.x + x * ex, xey = centerA.y + x * ey, yex = y * ex, yey = y * ey;
474
475 pt1 = float2(xex - yey, xey + yex);
476 pt2 = float2(xex + yey, xey - yex);
477
478 return true;
479
480 }
481 else
482 {
483 // No Intersection, far outside or one circle within the other
484 pt1 = pt2 = Vector2.zero;
485 return false;
486 }
487 }

509 public static bool TriangleContainsXY(float3 pt, float3 a, float3 b, float3 c)
510 {
511 float A = 1 / 2 * (-b.y * c.x + a.y * (-b.x + c.x) + a.x * (b.y - c.y) + b.x * c.y);
512 float sign = A < 0 ? -1 : 1;
513 float s = (a.y * c.x - a.x * c.y + (c.y - a.y) * pt.x + (a.x - c.x) * pt.y) * sign;
514 float t = (a.x * b.y - a.y * b.x + (a.y - b.y) * pt.x + (b.x - a.x) * pt.y) * sign;
515
516 return s > 0 && t > 0 && (s + t) < 2 * A * sign;
517 }

519 public static bool TriangleContainsXZ(float3 pt, float3 a, float3 b, float3 c)
520 {
521 float A = 1 / 2 * (-b.z * c.x + a.z * (-b.x + c.x) + a.x * (b.z - c.z) + b.x * c.z);
522 float sign = A < 0 ? -1 : 1;
523 float s = (a.z * c.x - a.x * c.z + (c.z - a.z) * pt.x + (a.x - c.x) * pt.z) * sign;
524 float t = (a.x * b.z - a.z * b.x + (a.z - b.z) * pt.x + (b.x - a.x) * pt.z) * sign;
525
526 return s > 0 && t > 0 && (s + t) < 2 * A * sign;
527 }
